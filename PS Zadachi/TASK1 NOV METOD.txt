#include <stdio.h>
#include <stdlib.h>
void BCDConversion(int n, int negative) {
    if (n == 0) {
        printf("0000");
        return;
    }

    int bcd = 0;
    int count = 0;

    while (n > 0) {
        int digit = n % 10;
        bcd = (bcd << 4) | digit;
        n /= 10;
        count++;
    }

    if(negative){
        printf("1001 ");
    }
    for (int i = count - 1; i >= 0; --i) {
        int nibble = (bcd >> (4 * i)) & 0xF;
        printf("%d%d%d%d ", (nibble & 8) >> 3, (nibble & 4) >> 2, (nibble & 2) >> 1, nibble & 1);
    }
}

int BCDAddition(int a, int b) {

    int result = 0;
    int negative = 0;

    if ((a < 0) && (abs(a) > b)){
        negative = 1;
    }
    else if((b < 0) && (abs(b) > a)){
        negative = 1;
    }

    result = a + b;


    printf("\nBCD Sum: ");
    BCDConversion(abs(result), negative);

    return 0;
}

int main() {
    int num1, num2;

    printf("first decimal number: ");
    scanf("%d", &num1);

    printf("second decimal number: ");
    scanf("%d", &num2);

    BCDAddition(num1, num2);

//   int result = num1 + num2;
